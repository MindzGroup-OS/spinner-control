Demo: Time Spinner
Demo Description: You can provide a min and max value with an interval for a range type of Spinner Control. In this demo below we are going to demonstrate the jQuery Spinner Control to spin Time.
Code: Time Spinner 
$(function () {
    $("#txtHours").SpinnerControl({
        type: 'range',
        typedata: { min: 0, max: 23, interval: 1 },
        defaultVal: 10,
        width: '50px'
    });
    $("#txtMinutes").SpinnerControl({
        type: 'range',
        typedata: { min: 0, max: 59, interval: 1 },
        defaultVal: 10,
        width: '50px'
    });
})
